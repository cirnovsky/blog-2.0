inline void mopai(int fuck)

inline void nmrqq(int fuck)

	for (int shit = A[fuck]._next; shit != fuck; shit = A[shit]._next)

inline void jisha(int Killer, int GGer)

inline void jdddd(int Killer, int GGer)

	}
}
